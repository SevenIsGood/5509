504  CSL_FEXT (hLcdc->regs->LCDLIDDCR, LCDC_LCDLIDDCR_LIDD_DMA_EN);

619 CSL_Status LCD_getConfigDMA (
620  CSL_LcdcHandle hLcdc,
621  CSL_LcdcConfigDma* pgetConfigDma )
622 {
623  Uint32 tmpAddr;
624  Uint16 tmpAddrLSB;
625  Uint16 tmpAddrMSB;
626 
627  if((hLcdc != NULL) && (pgetConfigDma != NULL))
628  {
629  /* passed parameters are ok */

631  else

634  {
635  return CSL_ESYS_BADHANDLE;
636  }
637  else
638  {
639  return CSL_ESYS_INVPARAMS;
640  }
641  }
642 
643  pgetConfigDma->burstSize = (CSL_LcdcDmaBurst)
644  CSL_FEXT (hLcdc->regs->LCDDMACR, LCDC_LCDDMACR_BURST_SIZE);
645 
646  pgetConfigDma->eofIntEn = (CSL_LcdcEofIntCtl)
647  CSL_FEXT (hLcdc->regs->LCDDMACR, LCDC_LCDDMACR_EOF_INTEN);
648 
649  pgetConfigDma->bigEndian = (CSL_LcdcEndianess)
650  CSL_FEXT (hLcdc->regs->LCDDMACR, LCDC_LCDDMACR_BIGENDIAN);
651 
652  pgetConfigDma->frameMode = (CSL_LcdcFrameMode)
653  CSL_FEXT (hLcdc->regs->LCDDMACR, LCDC_LCDDMACR_FRAME_MODE);
654 
655  /* DMA frame buffer 0 Base Address Set */
656  tmpAddrLSB = hLcdc->regs->LCDDMAFB0BAR0;
657  tmpAddrMSB = hLcdc->regs->LCDDMAFB0BAR1;
658  tmpAddr = tmpAddrMSB;
659  tmpAddr = tmpAddr << CSL_LCD_UINT16_NUMBER_BITS;
660  tmpAddr |= tmpAddrLSB;
661  tmpAddr = ((tmpAddr - CSL_LCD_SARAM_DMA_ADDR_OFFSET) >> CSL_LCD_DMA_SHIFT);
662  pgetConfigDma->fb0Base = tmpAddr;
663 
664  /* DMA frame buffer 0 Ceil Address Set*/

666  tmpAddrMSB = hLcdc->regs->LCDDMAFB0CAR1;
667  tmpAddr = tmpAddrMSB;

669  tmpAddr |= tmpAddrLSB;
670  tmpAddr = ((tmpAddr - CSL_LCD_SARAM_DMA_ADDR_OFFSET) >> CSL_LCD_DMA_SHIFT);
671  pgetConfigDma->fb0Ceil = tmpAddr;
672 
673  /* DMA frame buffer 1 Base Address Set */
674  tmpAddrLSB = hLcdc->regs->LCDDMAFB1BAR0;
675  tmpAddrMSB = hLcdc->regs->LCDDMAFB1BAR1;
676  tmpAddr = tmpAddrMSB;
677  tmpAddr = tmpAddr << CSL_LCD_UINT16_NUMBER_BITS;
678  tmpAddr |= tmpAddrLSB;
