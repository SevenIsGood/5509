187

197

339  pCtrlHandle = &pHIDClassHdl->ctrlHandle;
340  destPtr = NULL;
341 
342  switch(usbSetup->wValue >> CSL_HID_8BIT_SHIFT)
343  {
344  case CSL_USB_DEVICE_DESCRIPTOR_TYPE:
345  Desc_length = pCtrlHandle->deviceDescr[1]&0xFF;
346  destPtr = (char *)&pCtrlHandle->deviceDescr[1];
347  inOutFlag = CSL_USB_IOFLAG_NONE;
348  break;
349

353  destPtr = (char *)&pCtrlHandle->confDescr[10];
354  inOutFlag = CSL_USB_IOFLAG_NONE;
355 
356  if(Desc_length == usbSetup->wLength)
357  {
358  inOutFlag |= CSL_USB_IOFLAG_NOSHORT;
359  }
360  break;
361 
362  case CSL_USB_CONFIGURATION_DESCRIPTOR_TYPE:
363  inOutFlag = (CSL_USB_IOFLAG_LNK | CSL_USB_IOFLAG_CAT);
364  Desc_length = pCtrlHandle->confDescr[2]&0xFF;
365  destPtr = (char *)&pCtrlHandle->confDescr[1];
366 
367  if(Desc_length == usbSetup->wLength)
368  {
369  inOutFlag |= CSL_USB_IOFLAG_NOSHORT;
370  }
371  break;
372 
373  case CSL_USB_OTHERSPEED_CFG_DESCRIPTOR_TYPE:
374  inOutFlag = (CSL_USB_IOFLAG_LNK | CSL_USB_IOFLAG_CAT);
375  Desc_length = pCtrlHandle->osConfDescr[2]&0xFF;
376  destPtr = (char *)&pCtrlHandle->osConfDescr[1];
377 
378  if(Desc_length == usbSetup->wLength)
379  {
380  inOutFlag |= CSL_USB_IOFLAG_NOSHORT;
381  }
382  break;
383 
384  case CSL_USB_STRING_DESCRIPTOR_TYPE:
385 // if ((usbSetup->wValue&0x00FF)>3)
386 // return(CSL_HID_REQUEST_STALL);
387 
388  /* strlen of the string requested */
389  Desc_length = pCtrlHandle->strDescr[usbSetup->wValue&0x00FF][0]&0x00FF;
390  destPtr = (char *)&(pCtrlHandle->strDescr[usbSetup->wValue&0x00FF][0]);
391  inOutFlag = CSL_USB_IOFLAG_NONE;
392  break;
393 
394  case CSL_USB_INTERFACE_DESCRIPTOR_TYPE:
395  inOutFlag = (CSL_USB_IOFLAG_LNK | CSL_USB_IOFLAG_CAT);
396  Desc_length = pCtrlHandle->confDescr[9]&0xFF;
397  destPtr = (char *)&pCtrlHandle->confDescr[9];
398  inOutFlag = CSL_USB_IOFLAG_NONE;
399

403  }
404  break;
405 
406  case CSL_USB_ENDPOINT_DESCRIPTOR_TYPE:
407  inOutFlag = (CSL_USB_IOFLAG_LNK | CSL_USB_IOFLAG_CAT);
408  Desc_length = pCtrlHandle->confDescr[18]&0xFF;
409  destPtr = (char *)&pCtrlHandle->confDescr[18];
410  inOutFlag = CSL_USB_IOFLAG_NONE;
411 
412  if(Desc_length == usbSetup->wLength)
413  {
414  inOutFlag |= CSL_USB_IOFLAG_NOSHORT;
415  }
416  break;
417 
418  case CSL_USB_DEVICE_QUAL_DESCRIPTOR_TYPE:
419  Desc_length = pCtrlHandle->deviceQualifierDescr[1] & 0xFF;
420  destPtr = (char *)&pCtrlHandle->deviceQualifierDescr[1];
421  inOutFlag = CSL_USB_IOFLAG_NONE;
422  break;
423 
424  case CSL_USB_HID_REPORT_DESCRIPTOR_TYPE:
425  Desc_length = pCtrlHandle->hidClassHidReportDescrLen;
426  destPtr = (char *)&pCtrlHandle->hidClassHidReportDescr[1];
427  inOutFlag = CSL_USB_IOFLAG_NONE;
428  break;
429 
430  default:
431  return(CSL_HID_REQUEST_STALL);
432  }
433 
434  if (Desc_length == 0)
435  {
436  return(CSL_HID_REQUEST_STALL);
437  }
438  else if ((usbSetup->wLength <= (Uint16)Desc_length) && (Desc_length <= 64))
439  {
440  USB_postTransaction(hInEp, usbSetup->wLength,
441  (void*)destPtr,
442  CSL_USB_IN_TRANSFER);
443  }
444  else if (usbSetup->wLength > ((Uint16)(Desc_length)) && Desc_length <= 64)
445  {
446  USB_postTransaction(hInEp, Desc_length,
447  (void*)destPtr,
448  CSL_USB_IN_TRANSFER);
449  }
450  else if (usbSetup->wLength <= 64 && Desc_length > 64)
451  {
452  USB_postTransaction(hInEp, usbSetup->wLength,
453  (void*)destPtr,
454  CSL_USB_IN_TRANSFER);
455  }
456  else if ((usbSetup->wLength > 64) && (Desc_length > 64))
457  {
458  incrPtr = 0;
459  USB_postTransaction(hInEp, 64,
460  (void*)destPtr,
461  CSL_USB_IN_TRANSFER);
462 
463  if (usbSetup->wLength > ((Uint16)Desc_length))
464  {
465  Desc_length = Desc_length - 64;
466  }
467  else
468  {
469  Desc_length = usbSetup->wLength - 64;
470  }
471 
472  incrPtr = 64;
473 
474  while (Desc_length > 0)
475  {
476  if (Desc_length > 64)
477  {
478  USB_postTransaction(hInEp, 64,
479  (void*)(destPtr + incrPtr),
480  CSL_USB_IN_TRANSFER);
481  incrPtr += 64;
482  Desc_length -= 64;
483  }
484  else
485  {
486  USB_postTransaction(hInEp, Desc_length,
487  (void*)(destPtr + incrPtr),
488  CSL_USB_IN_TRANSFER);
489  Desc_length = 0;
490  }
491  }
492  }
493  else
494  {
495  return(CSL_HID_REQUEST_STALL);
496  }
497 
498  return(CSL_HID_REQUEST_GET_ACK);
499 }

728 CSL_HidRequestRet HID_reqGetStatus(CSL_UsbDevHandle devHandle,
729  CSL_UsbSetupStruct *usbSetup,

731  CSL_UsbEpHandle hOutEp,
732  void *pHidObj)

736  CSL_HidCtrlObject *pCtrlHandle;
737  CSL_UsbEpHandle hEPx;
738  CSL_Status status;
739 
740  pHidClassHdl = (pHidClassHandle)(pHidObj);
741  pCtrlHandle = &pHidClassHdl->ctrlHandle;
742  retStat = CSL_HID_REQUEST_GET_ACK;
743 
744  pCtrlHandle->ctrlBuffer[0] = 0x00;
745 
746  switch(usbSetup->bmRequestType - CSL_HID_REQUEST_TYPE_BASE)
747  {
748  /* Device Status to be returned */
749  case CSL_HID_REQUEST_TYPE_DEVICE_STATUS:
750  pCtrlHandle->ctrlBuffer[0] =
751  (((Uint16)USB_getRemoteWakeupStat(devHandle))<<1) |
752  0;//CSL_HID_CURRDEV_STAT;
753  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
754  CSL_USB_IOFLAG_NONE);
755  break;
756 
757  /* Interface status is to be returned */
758  case CSL_HID_REQUEST_TYPE_INTERFACE_STATUS:
759  //pCtrlHandle->ctrlBuffer[1] = CSL_HID_CURRDEV_STAT;
760  pCtrlHandle->ctrlBuffer[0] = 0x0; /* request to interface returns 0 */
761  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
762  CSL_USB_IOFLAG_NONE);
763  break;
764 
765  /* Endpoint status to be returned */
766  case CSL_HID_REQUEST_TYPE_EP_STATUS:
767  hEPx = devHandle->hEpHandleArray[3];
768  pCtrlHandle->ctrlBuffer[0] = (Uint16)USB_getEndptStall(hEPx,
769  &status);
770  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
771  CSL_USB_IOFLAG_NONE);
772  break;
773 
774  default:
775  /* STALL the endpoint */
776  retStat = CSL_HID_REQUEST_STALL;
777  break;
778  }
779 
780  return(retStat);
