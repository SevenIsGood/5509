186 static inline
187 fpAC_REQ_HANDLER AC_lookupReqHandler(Uint16 request,
188  CSL_AcRequestStruct *pUSB_ReqTable)
189 {
190  Uint16 index;
191 
192  /* parse thru the end of request handler table */
193  for(index = 0; pUSB_ReqTable[index].fpRequestHandler != NULL; index++)
194  {
195  /* if request handler exists return a pointer to the request handler routine */
196  if(pUSB_ReqTable[index].request == request)
197  {
198  return(pUSB_ReqTable[index].fpRequestHandler);
199  }
200  }
201 
202  /* if request handler does not exist return a pointer to the USB_reqUnknown
203  routine */
204  return(AC_reqUnknown);
205 }

500 static inline
501 CSL_AcRequestRet AC_reqGetStatus(CSL_UsbDevHandle devHandle,
502  CSL_UsbSetupStruct *usbSetup,
503  CSL_UsbEpHandle hInEp,
504  CSL_UsbEpHandle hOutEp,
505  void *pAcObj)
506 {
507  CSL_AcRequestRet retStat;
508  pAcClassHandle pAcClassHdl;
509  CSL_AcCtrlObject *pCtrlHandle;
510  CSL_UsbEpHandle hEPx;
511  CSL_Status status;
512  Uint16 endpt; /* this is USB logical endpoint */
513 
514  pAcClassHdl = (pAcClassHandle)(pAcObj);
515  pCtrlHandle = &pAcClassHdl->ctrlHandle;
516  retStat = CSL_AC_REQUEST_GET_ACK;
517 
518  switch(usbSetup->bmRequestType - CSL_AC_REQUEST_TYPE_BASE)
519  {
520  /* Device Status to be returned */
521  case CSL_AC_REQUEST_TYPE_DEVICE_STATUS:
522  pCtrlHandle->ctrlBuffer[1] =
523  (((Uint16)USB_getRemoteWakeupStat(devHandle))<<1) |
524  CSL_AC_CURRDEV_STAT;
525  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
526  CSL_USB_IOFLAG_NONE);
527  break;
528 
529  /* Interface status is to be returned */
530  case CSL_AC_REQUEST_TYPE_INTERFACE_STATUS:
531  pCtrlHandle->ctrlBuffer[1] = CSL_AC_CURRDEV_STAT;
532  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
533  CSL_USB_IOFLAG_NONE);
534  break;
535 
536  /* Endpoint status to be returned */
537  case CSL_AC_REQUEST_TYPE_EP_STATUS:
538 
539  endpt = usbSetup->wIndex & 0xFF;
540  hEPx = USB_epNumToHandle(devHandle, endpt);
541  pCtrlHandle->ctrlBuffer[1] = (Uint16)USB_getEndptStall(hEPx,
542  &status);
543  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
544  CSL_USB_IOFLAG_NONE);
545  break;
546 
547  default:
548  /* STALL the endpoint */
549  retStat = CSL_AC_REQUEST_STALL;
550  break;
551  }
552 
553  return(retStat);
554 }
