1098 static inline
1099 Int DMA_getStatus (
1103  CSL_DMA_Handle hDMA
1104 )
1105 {
1106  Int status;
1107  Uint16 chanNum;
1108 
1109  if(hDMA != NULL)
1110  {
1111  chanNum = hDMA->chanNum;
1112 
1113  while(chanNum >= CSL_DMA_PER_CNT)
1114  {
1115  chanNum = chanNum - CSL_DMA_PER_CNT;
1116  }
1117 
1118  switch((CSL_DMAChanNum)chanNum)
1119  {
1120 #if (defined(CHIP_C5517))
1121  case CSL_DMA_CHAN0:
1122  status = CSL_FEXT(hDMA->dmaRegs->DMACH0TCRU, DMA_DMACH0TCRU_EN);
1123  break;
1124  case CSL_DMA_CHAN1:
1125  status = CSL_FEXT(hDMA->dmaRegs->DMACH1TCRU, DMA_DMACH1TCRU_EN);
1126  break;
1127  case CSL_DMA_CHAN2:
1128  status = CSL_FEXT(hDMA->dmaRegs->DMACH2TCRU, DMA_DMACH2TCRU_EN);
1129  break;
1130  case CSL_DMA_CHAN3:
1131  status = CSL_FEXT(hDMA->dmaRegs->DMACH3TCRU, DMA_DMACH3TCRU_EN);
1132  break;
1133 #else
1134  case CSL_DMA_CHAN0:
1135  status = CSL_FEXT(hDMA->dmaRegs->DMACH0TCR2, DMA_DMACH0TCR2_EN);
1136  break;
1137  case CSL_DMA_CHAN1:
1138  status = CSL_FEXT(hDMA->dmaRegs->DMACH1TCR2, DMA_DMACH1TCR2_EN);
1139  break;
1140  case CSL_DMA_CHAN2:
1141  status = CSL_FEXT(hDMA->dmaRegs->DMACH2TCR2, DMA_DMACH2TCR2_EN);
1142  break;
1143  case CSL_DMA_CHAN3:
1144  status = CSL_FEXT(hDMA->dmaRegs->DMACH3TCR2, DMA_DMACH3TCR2_EN);
1145  break;
1146 #endif
1147  }
1148  }
1149  else
1150  {
1151  status = CSL_ESYS_BADHANDLE;
1152  }
