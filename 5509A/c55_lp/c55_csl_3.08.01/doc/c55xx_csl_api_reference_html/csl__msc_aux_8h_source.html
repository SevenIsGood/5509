228 static inline
229 CSL_Status MSC_HandleStartStopUnit(CSL_MscObject *pMscHandle,
230  CSL_UsbEpHandle hUsbInEp,
231  Uint16 lunNum)
232 {
233  CSL_MscMediaStatus mediaEjectStat;
234  CSL_Status status;
235  Uint16 loadEject;
236 
237  if (pMscHandle->cbwDataTransferLength != 0)
238  {
239  /* Stall the end point */
240  pMscHandle->storageState = CSL_MSC_STORAGE_STATE_SENDING_STALL;
241  status = USB_stallEndpt(pMscHandle->bulkOutEpHandle);
242  status = USB_stallEndpt(pMscHandle->bulkInEpHandle);
243 
244 
245  /* Dummy data transfer to intimate MSC_bulk function */
246  status |= USB_postTransaction(pMscHandle->bulkInEpHandle,
247  0,
248  pMscHandle->lbaBuffer,
249  CSL_USB_IOFLAG_NOT);
250 
251  return(status);
252  }
253 
254  loadEject = pMscHandle->cbw[CSL_MSC_CBW_9] >> CSL_MSC_8BIT_SHIFT;
255  pMscHandle->csw[CSL_MSC_CSW_6] = CSL_MSC_CSW_STATUS_COMMAND_PASSED;
256 
257  pMscHandle->storageState = CSL_MSC_STORAGE_STATE_SENDING_CSW;
258 
259  if(loadEject == 0x02) /* If the Eject Command */
260  {
261  if (pMscHandle->lun[lunNum].mediaState & CSL_MSC_MEDIA_LOCKED)
262  {
263  MSC_SetSenseKeys(pMscHandle->senseData,
264  CSL_MSC_SCSI_SENSEKEY_UNIT_ATTENTION,
265  CSL_MSC_SCSI_ASC_MEDIA_REMOVAL_PREVENTED);
266  pMscHandle->csw[CSL_MSC_CSW_6] = CSL_MSC_CSW_STATUS_COMMAND_FAILED;
267  }
268  else
269  {
270  mediaEjectStat = pMscHandle->mediaEject(lunNum);
271  if (mediaEjectStat == CSL_MSC_MEDIACCESS_SUCCESS)
272  {
273  /* Change the media state to not present or not bad */
274  pMscHandle->lun[lunNum].mediaState &=~(CSL_MSC_MEDIA_PRESENT | CSL_MSC_MEDIA_BAD);
275  MSC_SetSenseKeys(pMscHandle->senseData,
276  CSL_MSC_SCSI_SENSEKEY_NO_SENSE,
277  CSL_MSC_SCSI_ASC_NO_ADDITIONAL_SENSE_INFORMATION);
278  }
279  else
280  {
281  pMscHandle->csw[CSL_MSC_CSW_6] = CSL_MSC_CSW_STATUS_COMMAND_FAILED;
282  MSC_SetSenseKeys(pMscHandle->senseData,
283  CSL_MSC_SCSI_SENSEKEY_ILLEGAL_REQUEST,
284  CSL_MSC_SCSI_ASC_INVALID_COMMAND_OPERATION_CODE);
285  }
286  }
287  }
288  else
289  {
290  MSC_SetSenseKeys(pMscHandle->senseData,
291  CSL_MSC_SCSI_SENSEKEY_NO_SENSE,
292  CSL_MSC_SCSI_ASC_NO_ADDITIONAL_SENSE_INFORMATION);
293  /* Change the media state to not present or not bad .. PRASAD*/
294  pMscHandle->lun[lunNum].mediaState &=~(CSL_MSC_MEDIA_PRESENT | CSL_MSC_MEDIA_BAD);
295  }
296 
297  status = USB_postTransaction(hUsbInEp,13,&pMscHandle->csw[0],CSL_USB_IOFLAG_NONE);
298 
299  return(status);
300 }

645  CSL_USB_IOFLAG_NOSHORT);
646 
647  pMscHandle->storageState = CSL_MSC_STORAGE_STATE_SENDING_DATA;
648  }

650  {
651  status = MSC_sendCswWithPhaseError(pMscHandle, hUsbInEp);
652  }
653 
654  return(status);
