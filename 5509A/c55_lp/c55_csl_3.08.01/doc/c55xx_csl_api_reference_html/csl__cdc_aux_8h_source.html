360 static inline
361 CSL_CdcRequestRet CDC_reqGetStatus(CSL_UsbDevHandle devHandle,
362  CSL_UsbSetupStruct *usbSetup,
363  CSL_UsbEpHandle hInEp,
364  CSL_UsbEpHandle hOutEp,
365  void *pCdc)
366 {
367  CSL_CdcRequestRet retStat;
368  pCdcClassHandle pCdcClassHdl;
369  CSL_CdcCtrlObject *pCtrlHandle;
370  CSL_UsbEpHandle hEPx;
371  CSL_Status status;
372  Uint16 endpt; /* this is USB logical endpoint */
373 
374  pCdcClassHdl = (pCdcClassHandle)(pCdc);
375  pCtrlHandle = &pCdcClassHdl->ctrlHandle;
376  retStat = CSL_CDC_REQUEST_GET_ACK;
377 
378  switch(usbSetup->bmRequestType - CSL_CDC_REQUEST_TYPE_BASE)
379  {
380  /* Device Status to be returned */
381  case CSL_CDC_REQUEST_TYPE_DEVICE_STATUS:
382  pCtrlHandle->ctrlBuffer[0] =
383  (((Uint16)USB_getRemoteWakeupStat(devHandle))<<1) |
384  CSL_CDC_CURRDEV_STAT;
385  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
386  CSL_USB_IOFLAG_NONE);
387  break;
388 
389  /* Interface status is to be returned */
390  case CSL_CDC_REQUEST_TYPE_INTERFACE_STATUS:
391  pCtrlHandle->ctrlBuffer[0] = CSL_CDC_CURRDEV_STAT;
392  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
393  CSL_USB_IOFLAG_NONE);
394  break;
395 
396  /* Endpoint status to be returned */
397  case CSL_CDC_REQUEST_TYPE_EP_STATUS:
398 
399  endpt = usbSetup->wIndex & 0xFF;
400  hEPx = USB_epNumToHandle(devHandle, endpt);
401  pCtrlHandle->ctrlBuffer[0] = (Uint16)USB_getEndptStall(hEPx,
402  &status);
403  USB_postTransaction(hInEp, 2, &pCtrlHandle->ctrlBuffer,
404  CSL_USB_IOFLAG_NONE);
405  break;
406 
407  default:
408  /* STALL the endpoint */
409  retStat = CSL_CDC_REQUEST_STALL;
410  break;
411  }
412 
413  return(retStat);
414 }
